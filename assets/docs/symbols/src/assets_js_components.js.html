227     let format = typeof args === 'object' ? args.format : args
228     const now = new Date();
229     let dd = now.getDate();
230     let mm = now.getMonth() + 1; 
231   
232     const yyyy = now.getFullYear();
233     if (dd < 10) dd = '0' + dd;
234     if (mm < 10) mm = '0' + mm;
235     format = format ? format : 'mm-dd-yyyy';
236     format = format.replace('dd',dd).replace('mm',mm).replace('yyyy',yyyy);
237     if(typeof args.modify === 'function') format = args.modify(format)
238     if(args.el){
239       componentElementOutput({
240         el : args.el,
241         output : format
242       });
243     }
244    
245     return format;

360   function api(args,callback){
361     if(!args.method) args.method = 'get';
362 
363     axios[args.method](args.url,args.data)
364     .then((res) => {
365       const data = [];
366       if(res.data.length > 0){
367         for(let item of res.data){
368           if(args.modify) item = args.modify(item);
369           data.push(item);
370           
371         }
372       }
373       if(callback) 
374             return callback(data);
375           if(args.callback) 
376             return args.callback(data);
377       
378           return data;
379       
380     })

489       const name = args[0].split(':')[1]

491       const attrs = args[1]
492       if(attrs.appendTo){
493         const appendTo = document.querySelectorAll(attrs.appendTo)
494         appendTo.appendChild(element)
495         delete attrs.appendTo
496       }
497       if(attrs.component){
498         if(args.protoype === Array){
499          component[attrs.component[0]](attrs.component[1])
500         }
501       }
502       
503       Object.getOwnPropertyNames(attrs).map((attr)=>{
504         element.setAttribute(attr,attrs[attr]);
505       });
506       if(args[2].protoype === Array  ){
507         element.appendChild(el(args[2]));
508       }else if(typeof args[2] === 'object'){
509         element.appendChild(args[2]);
510       }else{
511         element.innerHTML = args[2];
512       }
513       componentMethods({
514         element : element,
515         name : name
516       });
517       return element
